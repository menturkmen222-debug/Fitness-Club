## 🎯 NÄMÄ GURÝARSYŇ

Fitnes klubunyň dolandyryjylary, tälimçiler we administrator üçin  
**doly, professional admin paneli** — `admin.html` ýeke faýlda.

Öňki döredilen `index.html` fitnes saýtynyň ähli maglumatlary —  
agzalar, sapaklar, tälimçiler, abunalar, maliýe — hemmesi şu panelden dolandyrylýar.

**Hiç hili backend ýok. `localStorage` esasynda. Real işe goýlanda backend bilen çalşylmaga taýýar arhitektura.**

---

## 🛠️ TEHNIKI GURLUŞ

```
Faýl:         admin.html  (HTML + CSS + JS — bir ýerde)
Maglumat:     localStorage  (JSON, doly CRUD)

Şrift:        Google Fonts — Outfit + Space Mono (CDN)
Ikonlar:      Lucide Icons (CDN)
```

```javascript
function detectScheduleConflicts(classes) {
  const conflicts = [];
  classes.forEach((c1, i) => {
    classes.forEach((c2, j) => {
      if (i >= j) return;
      if (c1.hallId !== c2.hallId) return;
      
      const sharedDays = c1.dayOfWeek.filter(d => c2.dayOfWeek.includes(d));
      if (!sharedDays.length) return;
      
      const c1End = addMinutes(c1.startTime, c1.duration);
      const c2End = addMinutes(c2.startTime, c2.duration);
      
      if (c1.startTime < c2End && c1End > c2.startTime) {
        conflicts.push({ class1: c1, class2: c2, days: sharedDays });
      }
    });
  });
  return conflicts;
}
// Çaknyşma tapylsa — tertipde gyzyl bilen bellenilýär + habar
```

```javascript
const DB = {
  get:    (key) => {
    try { return JSON.parse(localStorage.getItem('gym_' + key)) || []; }
    catch { return []; }
  },
  set:    (key, data) => localStorage.setItem('gym_' + key, JSON.stringify(data)),
  add:    (key, item) => {
    const list = DB.get(key);
    const newItem = {
      ...item,
      id: 'ID-' + Date.now(),
      createdAt: new Date().toISOString()
    };
    list.push(newItem);
    DB.set(key, list);
    return newItem;
  },
  update: (key, id, patch) => {
    const list = DB.get(key).map(i =>
      i.id === id ? { ...i, ...patch, updatedAt: new Date().toISOString() } : i
    );
    DB.set(key, list);
    return list;
  },
  remove: (key, id) => {
    const list = DB.get(key).filter(i => i.id !== id);
    DB.set(key, list);
    return list;
  },
  find:   (key, id) => DB.get(key).find(i => i.id === id),
  filter: (key, fn) => DB.get(key).filter(fn),
  count:  (key, fn) => fn ? DB.get(key).filter(fn).length : DB.get(key).length,
};
```

```javascript
const toast = {
  show(msg, type = 'success', duration = 3500) {
    const el = document.createElement('div');
    el.className = `toast toast-${type}`;
    el.innerHTML = `
      <span class="toast-icon">${{success:'✅',error:'❌',warning:'⚠️',info:'ℹ️'}[type]}</span>
      <span class="toast-msg">${msg}</span>
      <button class="toast-close" onclick="this.parentElement.remove()">✕</button>
    `;
    document.getElementById('toast-container').appendChild(el);
    setTimeout(() => el.classList.add('toast-exit'), duration);
    setTimeout(() => el.remove(), duration + 350);
  },
  success: (msg) => toast.show(msg, 'success'),
  error:   (msg) => toast.show(msg, 'error'),
  warning: (msg) => toast.show(msg, 'warning'),
  info:    (msg) => toast.show(msg, 'info'),
};
```

```javascript
function exportCSV(data, filename) {
  if (!data.length) return toast.warning('Maglumat ýok');
  const headers = Object.keys(data[0]);
  const rows = data.map(r => headers.map(h => `"${r[h] ?? ''}"`).join(','));
  const csv = [headers.join(','), ...rows].join('\n');
  downloadFile(csv, filename + '.csv', 'text/csv');
}

function exportJSON(data, filename) {
  downloadFile(JSON.stringify(data, null, 2), filename + '.json', 'application/json');
}

function downloadFile(content, filename, type) {
  const a = document.createElement('a');
  a.href = URL.createObjectURL(new Blob([content], { type }));
  a.download = filename;
  a.click();
}
```
